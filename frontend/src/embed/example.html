Usage: CDN Script Tag

<!-- Include the widget script -->
<script src="https://cdn.example.com/keepsave-widget.umd.js"></script>

<!-- Use the widget with direct API key auth -->
<keepsave-widget
  project-id="your-project-id"
  api-url="https://your-keepsave-server.com"
  api-key="ks_your_api_key"
  theme="light"
  mode="read"
></keepsave-widget>

Usage: postMessage Auth

<!-- Widget without direct credentials -->
<keepsave-widget
  id="my-widget"
  project-id="your-project-id"
  api-url="https://your-keepsave-server.com"
  theme="light"
  mode="readwrite"
></keepsave-widget>

<script>
  // Listen for the widget's auth request
  window.addEventListener('message', (event) => {
    if (event.data.type === 'keepsave-auth-request') {
      // Respond with credentials
      event.source.postMessage({
        type: 'keepsave-auth',
        token: 'your-jwt-token'
        // or: apiKey: 'ks_your_api_key'
      }, '*');
    }
  });
</script>

Usage: ES Module Import

import { register, KeepSaveAPI } from 'keepsave-widget';

// Auto-registers <keepsave-widget> custom element
register();

// Or use the API client directly
const api = new KeepSaveAPI('https://your-keepsave-server.com');
api.setApiKey('ks_your_api_key');

const secrets = await api.listSecrets('project-id', 'alpha');
